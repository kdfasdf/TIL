인접칸 조사 함수구현보다 그냥 구현 하는 것이 더 나았던 문제

```
#python
n=int(input())
data=[[0]*n for _ in range(n)]
students=[list(map(int,input().split())) for _ in range(n**2)]
dx=[1,-1,0,0]
dy=[0,0,1,-1]
for i in range(n**2):
    student=students[i][0]
    temp=[]
    for j in range(n):
        for k in range(n):
            if data[j][k]==0:
                like = 0
                blank = 0
                for l in range(4):
                    n_x=k+dx[l]
                    n_y=j+dy[l]
                    if 0<=n_x<n and 0<=n_y<n:
                        if data[n_y][n_x] in students[i][1:]:
                            like+=1
                        if data[n_y][n_x]==0:
                            blank+=1
                temp.append([like,blank,j,k])
    temp.sort(key=lambda x:(-x[0],-x[1],x[2],x[3]))
    data[temp[0][2]][temp[0][3]]=student
result=0
students.sort()

for i in range(n):
    for j in range(n):
        answer=0
        for k in range(4):
            n_y=i+dy[k]
            n_x=j+dx[k]
            if 0<=n_y<n and 0<=n_x<n:
                if data[n_y][n_x] in students[data[i][j]-1]:
                    answer+=1
        if answer!=0:
            result+=10**(answer-1)
print(result)

```
