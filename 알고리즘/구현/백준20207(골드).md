```
#python
n=int(input())
cal=[0]*366
for i in range(n):
    a,b=map(int,input().split())
    for j in range(a,b+1):
        cal[j]+=1
y=0
x=0
result=0
for i in range(366):
    if cal[i]!=0:
        y=max(y,cal[i])
        x+=1
    else:
        result+=x*y
        x=0
        y=0
result+=x*y
print(result)
```

```
//Java
import java.util.*;
public class Main{
     public static void main(String[] args){
         Scanner sc = new Scanner(System.in);
         int n=sc.nextInt();
         int[] cal  =new int[366];
         int result=0;
         int x =0;
         int y=0;

         for(int i=0; i<n;i++){
             int a=sc.nextInt();
             int b=sc.nextInt();
             for (int j=a;j<=b;j++)
             {
                 cal[j]+=1;
             }
         }

         for(int i=1;i<366;i++){
                if(cal[i]!=0)
                {
                    y=Math.max(y,cal[i]);
                    x+=1;
                }
                else{
                    result+=x*y;
                    x=0;
                    y=0;
                }
         }
         result+=x*y;
         System.out.println(result);
     }
 }
```
